# Plan 04-04: Explainer Enhancements — Passage-Level & History

**Objective:** Extend the explainer system to support passage-level generation from text selection, and add an explainer history list inside the existing ExplainerPanel with Current/History tab navigation.

**Must-haves for phase goal:**
- User can select text, click "Explain this to me," and receive a passage-level explainer
- Passage-level explainers use the same cache architecture as book/section explainers
- Explainer panel shows a "History" tab listing all explainers for the current book
- Each history entry shows type, target label, language, date, and tier
- History entries have "Go to context" navigation

```ts
import { NextResponse } from "next/server";
import { requireAuth } from "@/lib/auth-guards";
import { verifyBookAccess } from "@/server/services/reader";
import { db } from "@/server/db";

export async function GET(request: Request) {
  try {
    const user = await requireAuth();
    const { searchParams } = new URL(request.url);
    const bookId = searchParams.get("bookId");

    if (!bookId) {
      return NextResponse.json({ error: "bookId is required" }, { status: 400 });
    }

    const hasAccess = await verifyBookAccess(user.id, bookId);
    if (!hasAccess) {
      return NextResponse.json({ error: "Access denied" }, { status: 403 });
    }

    const history = await db.explainerRequest.findMany({
      where: { userId: user.id, bookId },
      include: { explainer: true },
      orderBy: { createdAt: "desc" },
    });

    const formatted = history.map((req) => ({
      id: req.id,
      explainerId: req.explainerId,
      contentType: req.contentType,
      sectionHref: req.sectionHref,
      selectedTextPreview: req.selectedTextPreview,
      language: req.language,
      tier: req.tier,
      createdAt: req.createdAt,
      content: req.explainer.content,
      modelId: req.explainer.modelId,
    }));

    return NextResponse.json({ history: formatted });
  } catch (error: any) {
    if (error.statusCode === 401) {
      return NextResponse.json({ error: "Authentication required" }, { status: 401 });
    }
    console.error("[GET /api/explainers/history]", error);
    return NextResponse.json({ error: "Internal server error" }, { status: 500 });
  }
}
```

Also create a DELETE endpoint for removing history entries:

```ts
export async function DELETE(request: Request) {
  try {
    const user = await requireAuth();
    const { searchParams } = new URL(request.url);
    const requestId = searchParams.get("id");

    if (!requestId) {
      return NextResponse.json({ error: "id is required" }, { status: 400 });
    }

    await db.explainerRequest.deleteMany({
      where: { id: requestId, userId: user.id },
    });

    return NextResponse.json({ success: true });
  } catch (error: any) {
    if (error.statusCode === 401) {
      return NextResponse.json({ error: "Authentication required" }, { status: 401 });
    }
    console.error("[DELETE /api/explainers/history]", error);
    return NextResponse.json({ error: "Internal server error" }, { status: 500 });
  }
}
```

```tsx
"use client";

import { useState, useEffect, useCallback } from "react";
import { ScrollArea } from "@/components/ui/scroll-area";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import { Skeleton } from "@/components/ui/skeleton";
import { BookOpen, FileText, Highlighter, ArrowRight, Trash2 } from "lucide-react";
import { cn } from "@/lib/utils";
import { getLanguageName } from "@/lib/languages";

interface HistoryEntry {
  id: string;
  explainerId: string;
  contentType: "book" | "section" | "passage";
  sectionHref?: string;
  selectedTextPreview?: string;
  language: string;
  tier: string;
  createdAt: string;
  content: string;
  modelId: string;
}

interface ExplainerHistoryProps {
  bookId: string;
  onSelectEntry?: (entry: HistoryEntry) => void;
}

export function ExplainerHistory({ bookId, onSelectEntry }: ExplainerHistoryProps) {
  const [history, setHistory] = useState<HistoryEntry[]>([]);
  const [isLoading, setIsLoading] = useState(false);
  const [expandedId, setExpandedId] = useState<string | null>(null);

  const loadHistory = useCallback(async () => {
    setIsLoading(true);
    try {
      const res = await fetch(`/api/explainers/history?bookId=${bookId}`);
      if (res.ok) {
        const data = await res.json();
        setHistory(data.history || []);
      }
    } catch (err) {
      console.error("[ExplainerHistory] Failed to load:", err);
    } finally {
      setIsLoading(false);
    }
  }, [bookId]);

  useEffect(() => {
    loadHistory();
  }, [loadHistory]);

  const handleDelete = useCallback(async (id: string) => {
    try {
      await fetch(`/api/explainers/history?id=${id}`, { method: "DELETE" });
      setHistory((prev) => prev.filter((h) => h.id !== id));
    } catch (err) {
      console.error("[ExplainerHistory] Failed to delete:", err);
    }
  }, []);

  const getTypeIcon = (type: string) => {
    switch (type) {
      case "book": return <BookOpen className="h-3.5 w-3.5" />;
      case "section": return <FileText className="h-3.5 w-3.5" />;
      case "passage": return <Highlighter className="h-3.5 w-3.5" />;
      default: return <BookOpen className="h-3.5 w-3.5" />;
    }
  };

  const getTypeLabel = (type: string) => {
    switch (type) {
      case "book": return "Book";
      case "section": return "Section";
      case "passage": return "Passage";
      default: return type;
    }
  };

  if (isLoading) {
    return (
      <div className="p-4 space-y-3">
        {Array.from({ length: 5 }).map((_, i) => (
          <div key={i} className="space-y-2">
            <Skeleton className="h-4 w-24" />
            <Skeleton className="h-3 w-full" />
          </div>
        ))}
      </div>
    );
  }

  if (history.length === 0) {
    return (
      <div className="flex flex-col items-center justify-center h-full px-4 text-center">
        <Sparkles className="h-8 w-8 text-muted-foreground mb-3" />
        <p className="text-sm text-muted-foreground">
          No Explainers yet.
        </p>
        <p className="text-xs text-muted-foreground mt-1">
          Request an Explainer from the book page, table of contents, or by selecting text.
        </p>
      </div>
    );
  }

  return (
    <ScrollArea className="h-full">
      <div className="py-2">
        {history.map((entry) => (
          <div
            key={entry.id}
            className="px-4 py-3 border-b border-border/50 group"
          >
            <div className="flex items-center gap-2 mb-1">
              {getTypeIcon(entry.contentType)}
              <span className="text-xs font-medium">{getTypeLabel(entry.contentType)}</span>
              <Badge variant="secondary" className="text-[10px] h-5">
                {getLanguageName(entry.language)}
              </Badge>
              {entry.tier === "pro" && (
                <Badge variant="default" className="text-[10px] h-5">Pro</Badge>
              )}
              <span className="text-xs text-muted-foreground ml-auto">
                {new Date(entry.createdAt).toLocaleDateString()}
              </span>
            </div>

            {entry.selectedTextPreview && (
              <p className="text-xs text-muted-foreground line-clamp-2 mb-2">
                "{entry.selectedTextPreview}"
              </p>
            )}

            <div className="flex items-center gap-2">
              <Button
                variant="ghost"
                size="sm"
                className="h-7 text-xs gap-1"
                onClick={() => {
                  setExpandedId(expandedId === entry.id ? null : entry.id);
                  onSelectEntry?.(entry);
                }}
              >
                {expandedId === entry.id ? "Hide" : "View"}
                <ArrowRight className="h-3 w-3" />
              </Button>

              <Button
                variant="ghost"
                size="icon-sm"
                className="opacity-0 group-hover:opacity-100 transition-opacity h-7 w-7"
                onClick={() => handleDelete(entry.id)}
              >
                <Trash2 className="h-3 w-3 text-destructive" />
              </Button>
            </div>

            {expandedId === entry.id && (
              <div className="mt-3 p-3 bg-muted/50 rounded-md text-sm leading-relaxed">
                {entry.content}
              </div>
            )}
          </div>
        ))}
      </div>
    </ScrollArea>
  );
}
```
