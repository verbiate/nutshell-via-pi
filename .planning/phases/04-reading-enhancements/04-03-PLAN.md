```tsx
"use client";

import { useState, useCallback } from "react";
import { Bookmark } from "lucide-react";
import { Button } from "@/components/ui/button";
import { cn } from "@/lib/utils";

interface BookmarkButtonProps {
  bookId: string;
  currentCfi: string | null;
  paragraphIndex: number;
  selectedText?: string;
  onBookmarkToggle?: (isBookmarked: boolean) => void;
}

export function BookmarkButton({
  bookId,
  currentCfi,
  paragraphIndex,
  selectedText,
  onBookmarkToggle,
}: BookmarkButtonProps) {
  const [isBookmarked, setIsBookmarked] = useState(false);
  const [isLoading, setIsLoading] = useState(false);

  const toggleBookmark = useCallback(async () => {
    if (!currentCfi || isLoading) return;

    setIsLoading(true);
    try {
      if (isBookmarked) {
        await fetch(`/api/reader/bookmarks?bookId=${bookId}&cfi=${encodeURIComponent(currentCfi)}`, {
          method: "DELETE",
        });
        setIsBookmarked(false);
        onBookmarkToggle?.(false);
      } else {
        await fetch("/api/reader/bookmarks", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            bookId,
            cfi: currentCfi,
            paragraphIndex,
            selectedText: selectedText || "",
          }),
        });
        setIsBookmarked(true);
        onBookmarkToggle?.(true);
      }
    } catch (err) {
      console.error("[BookmarkButton] Failed to toggle bookmark:", err);
    } finally {
      setIsLoading(false);
    }
  }, [bookId, currentCfi, paragraphIndex, selectedText, isBookmarked, isLoading, onBookmarkToggle]);

  return (
    <Button
      variant="ghost"
      size="icon-sm"
      onClick={toggleBookmark}
      disabled={isLoading || !currentCfi}
      aria-label={isBookmarked ? "Remove bookmark" : "Add bookmark"}
      className={cn("shrink-0 transition-colors", isBookmarked && "text-primary")}
    >
      <Bookmark
        className={cn("h-4 w-4", isBookmarked && "fill-current")}
      />
    </Button>
  );
}
```

```tsx
"use client";

import { useState, useEffect, useCallback } from "react";
import {
  Sheet,
  SheetContent,
  SheetHeader,
  SheetTitle,
  SheetTrigger,
} from "@/components/ui/sheet";
import { ScrollArea } from "@/components/ui/scroll-area";
import { Button } from "@/components/ui/button";
import { Bookmark, Trash2 } from "lucide-react";
import { cn } from "@/lib/utils";

interface BookmarkItem {
  id: string;
  cfi: string;
  paragraphIndex: number;
  selectedText: string;
  note?: string;
  createdAt: string;
}

interface BookmarkPanelProps {
  bookId: string;
  onNavigate: (cfi: string) => void;
}

export function BookmarkPanel({ bookId, onNavigate }: BookmarkPanelProps) {
  const [bookmarks, setBookmarks] = useState<BookmarkItem[]>([]);
  const [isOpen, setIsOpen] = useState(false);
  const [isLoading, setIsLoading] = useState(false);

  const loadBookmarks = useCallback(async () => {
    setIsLoading(true);
    try {
      const res = await fetch(`/api/reader/bookmarks?bookId=${bookId}`);
      if (res.ok) {
        const data = await res.json();
        setBookmarks(data.bookmarks || []);
      }
    } catch (err) {
      console.error("[BookmarkPanel] Failed to load bookmarks:", err);
    } finally {
      setIsLoading(false);
    }
  }, [bookId]);

  useEffect(() => {
    if (isOpen) {
      loadBookmarks();
    }
  }, [isOpen, loadBookmarks]);

  const handleDelete = useCallback(
    async (cfi: string) => {
      try {
        await fetch(`/api/reader/bookmarks?bookId=${bookId}&cfi=${encodeURIComponent(cfi)}`, {
          method: "DELETE",
        });
        setBookmarks((prev) => prev.filter((b) => b.cfi !== cfi));
      } catch (err) {
        console.error("[BookmarkPanel] Failed to delete bookmark:", err);
      }
    },
    [bookId]
  );

  const handleNavigate = useCallback(
    (cfi: string) => {
      onNavigate(cfi);
      setIsOpen(false);
    },
    [onNavigate]
  );

  return (
    <Sheet open={isOpen} onOpenChange={setIsOpen}>
      <SheetTrigger asChild>
        <Button variant="ghost" size="icon-sm" aria-label="Open bookmarks" className="shrink-0">
          <Bookmark className="h-4 w-4" />
        </Button>
      </SheetTrigger>
      <SheetContent side="left" className="w-[320px] sm:w-[360px] p-0">
        <SheetHeader className="px-4 py-3 border-b border-border">
          <SheetTitle>Bookmarks</SheetTitle>
        </SheetHeader>
        <ScrollArea className="h-[calc(100vh-56px)]">
          <div className="py-2">
            {isLoading && (
              <div className="px-4 py-8 text-center text-sm text-muted-foreground">
                Loading bookmarks...
              </div>
            )}
            {!isLoading && bookmarks.length === 0 && (
              <div className="px-4 py-8 text-center text-sm text-muted-foreground">
                No bookmarks yet. Tap the bookmark button while reading to save your place.
              </div>
            )}
            {bookmarks.map((bookmark) => (
              <div
                key={bookmark.id}
                className="px-4 py-3 border-b border-border/50 group"
              >
                <button
                  onClick={() => handleNavigate(bookmark.cfi)}
                  className="w-full text-left"
                >
                  <p className="text-sm font-medium line-clamp-2">
                    {bookmark.selectedText || `Position ${bookmark.paragraphIndex}`}
                  </p>
                  {bookmark.note && (
                    <p className="text-xs text-muted-foreground mt-1">{bookmark.note}</p>
                  )}
                  <p className="text-xs text-muted-foreground mt-1">
                    {new Date(bookmark.createdAt).toLocaleDateString()}
                  </p>
                </button>
                <Button
                  variant="ghost"
                  size="icon-sm"
                  className="opacity-0 group-hover:opacity-100 transition-opacity mt-2"
                  onClick={() => handleDelete(bookmark.cfi)}
                  aria-label="Delete bookmark"
                >
                  <Trash2 className="h-3 w-3 text-destructive" />
                </Button>
              </div>
            ))}
          </div>
        </ScrollArea>
      </SheetContent>
    </Sheet>
  );
}
```

```tsx
  const [txtContent, setTxtContent] = useState("");
  const [isLoading, setIsLoading] = useState(false);
  const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  // Load TXT content when panel opens
  useEffect(() => {
    if (!isOpen || txtContent) return;

    async function loadTxt() {
      setIsLoading(true);
      try {
        const res = await fetch(`/api/books/${bookId}/txt`);
        if (res.ok) {
          const text = await res.text();
          setTxtContent(text);
        }
      } catch (err) {
        console.error("[SearchPanel] Failed to load TXT:", err);
      } finally {
        setIsLoading(false);
      }
    }

    loadTxt();
  }, [isOpen, bookId, txtContent]);

  // Search with debounce
  useEffect(() => {
    if (debounceRef.current) {
      clearTimeout(debounceRef.current);
    }

    if (!txtContent || query.length < 3) {
      setResults([]);
      return;
    }

    debounceRef.current = setTimeout(() => {
      const paragraphs = txtContent.split(/\n\n+/);
      const searchResults: SearchResult[] = [];
      const regex = new RegExp(`(${escapeRegExp(query)})`, "gi");

      paragraphs.forEach((paragraph, index) => {
        let match;
        while ((match = regex.exec(paragraph)) !== null) {
          const matchStart = match.index;
          const matchEnd = match.index + match[0].length;
          const snippetStart = Math.max(0, matchStart - 40);
          const snippetEnd = Math.min(paragraph.length, matchEnd + 40);
          const snippet = paragraph.slice(snippetStart, snippetEnd);

          searchResults.push({
            paragraphIndex: index,
            snippet: `...${snippet}...`,
            matchStart,
            matchEnd,
          });
        }
      });

      // Limit to first 50 results
      setResults(searchResults.slice(0, 50));
    }, 300);

    return () => {
      if (debounceRef.current) {
        clearTimeout(debounceRef.current);
      }
    };
  }, [query, txtContent]);

  const handleNavigate = useCallback(
    (paragraphIndex: number) => {
      onNavigate(paragraphIndex);
      setIsOpen(false);
    },
    [onNavigate]
  );

  const handleClear = useCallback(() => {
    setQuery("");
    setResults([]);
  }, []);

  return (
    <Sheet open={isOpen} onOpenChange={setIsOpen}>
      <SheetTrigger asChild>
        <Button variant="ghost" size="icon-sm" aria-label="Search in book" className="shrink-0">
          <Search className="h-4 w-4" />
        </Button>
      </SheetTrigger>
      <SheetContent side="left" className="w-[320px] sm:w-[360px] p-0">
        <SheetHeader className="px-4 py-3 border-b border-border">
          <SheetTitle>Search</SheetTitle>
        </SheetHeader>
        <div className="px-4 py-3 border-b border-border">
          <div className="relative">
            <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
            <Input
              placeholder="Search in book..."
              value={query}
              onChange={(e) => setQuery(e.target.value)}
              className="pl-9 pr-9"
            />
            {query && (
              <Button
                variant="ghost"
                size="icon-sm"
                className="absolute right-1 top-1/2 -translate-y-1/2"
                onClick={handleClear}
              >
                <X className="h-3 w-3" />
              </Button>
            )}
          </div>
        </div>
        <ScrollArea className="h-[calc(100vh-140px)]">
          <div className="py-2">
            {isLoading && (
              <div className="px-4 py-8 text-center text-sm text-muted-foreground">
                Loading book text...
              </div>
            )}
            {!isLoading && query.length > 0 && query.length < 3 && (
              <div className="px-4 py-8 text-center text-sm text-muted-foreground">
                Type at least 3 characters to search
              </div>
            )}
            {!isLoading && query.length >= 3 && results.length === 0 && (
              <div className="px-4 py-8 text-center text-sm text-muted-foreground">
                No results found for "{query}"
              </div>
            )}
            {results.map((result, idx) => (
              <button
                key={idx}
                onClick={() => handleNavigate(result.paragraphIndex)}
                className="w-full px-4 py-3 text-left border-b border-border/50 hover:bg-muted/50 transition-colors"
              >
                <p className="text-sm">
                  {result.snippet.split(new RegExp(`(${escapeRegExp(query)})`, "gi")).map((part, i) =>
                    part.toLowerCase() === query.toLowerCase() ? (
                      <mark key={i} className="bg-primary/20 rounded px-0.5">
                        {part}
                      </mark>
                    ) : (
                      <span key={i}>{part}</span>
                    )
                  )}
                </p>
                <p className="text-xs text-muted-foreground mt-1">
                  Paragraph {result.paragraphIndex}
                </p>
              </button>
            ))}
          </div>
        </ScrollArea>
      </SheetContent>
    </Sheet>
  );
}

function escapeRegExp(string: string): string {
  return string.replace(/[.*+?^${}()|[\]\\]/g, "\\$&");
}
```

```tsx
"use client";

import { useState, useEffect, useCallback } from "react";

interface Bookmark {
  id: string;
  cfi: string;
  paragraphIndex: number;
  selectedText: string;
  note?: string;
  createdAt: string;
}

interface Highlight {
  id: string;
  cfi: string;
  paragraphIndex: number;
  selectedText: string;
  color: string;
  note?: string;
  createdAt: string;
}

export function useReaderAnnotations(bookId: string) {
  const [bookmarks, setBookmarks] = useState<Bookmark[]>([]);
  const [highlights, setHighlights] = useState<Highlight[]>([]);
  const [isLoading, setIsLoading] = useState(false);

  const loadAnnotations = useCallback(async () => {
    setIsLoading(true);
    try {
      const [bookmarksRes, highlightsRes] = await Promise.all([
        fetch(`/api/reader/bookmarks?bookId=${bookId}`),
        fetch(`/api/reader/highlights?bookId=${bookId}`),
      ]);

      if (bookmarksRes.ok) {
        const data = await bookmarksRes.json();
        setBookmarks(data.bookmarks || []);
      }
      if (highlightsRes.ok) {
        const data = await highlightsRes.json();
        setHighlights(data.highlights || []);
      }
    } catch (err) {
      console.error("[useReaderAnnotations] Failed to load:", err);
    } finally {
      setIsLoading(false);
    }
  }, [bookId]);

  useEffect(() => {
    loadAnnotations();
  }, [loadAnnotations]);

  const refresh = useCallback(() => {
    loadAnnotations();
  }, [loadAnnotations]);

  return { bookmarks, highlights, isLoading, refresh };
}
```
